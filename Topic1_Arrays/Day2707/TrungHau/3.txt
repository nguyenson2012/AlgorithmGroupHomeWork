// Time complexity: O(n)
// Space complexity: O(1)
// Description: Creating a count variable to count the element. Because majority element appear more than n / 2.

class Solution {
public:
    int majorityElement(vector<int>& nums) {
        int res = 0, count = 0;
        for(int i = 0; i < nums.size(); i++){
            if(count > 0){
                if(nums[i] == res) count++;
                else count--;
            }
            else {
                res = nums[i];
                count = 1;
            }
        }
        return res;
    }
};

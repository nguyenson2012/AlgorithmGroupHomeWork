var merge = function (nums1, m, nums2, n) {
  let index1 = m - 1,
    index2 = n - 1,
    indexMerged = m + n - 1;

  while (index2 >= 0) {
    if (index1 >= 0 && nums1[index1] > nums2[index2]) {
      nums1[indexMerged--] = nums1[index1--];
    } else {
      nums1[indexMerged--] = nums2[index2--];
    }
  }

  return nums1;
};
